# LC35 

简单的一道二分查找

``` 
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var searchInsert = function(nums, target) {
    let [left, right] = [0, nums.length];
    while(left <= right) {
        let middle = parseInt(left + right);
        if(target === nums[middle]) return middle;
        else if(target > nums[middle]) left = middle + 1;
        else right = middle - 1;
    }
    return right + 1;    
};
```
