# LC34

知识点仍是二分查找，难点在于要单独进行target的左边界和有边界，当找到target时需要放大范围，若仍存在target，更新边界，若不存在，则不更新

```
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
function findBorder(border, nums, target) {
    let borderIndex = -1;
    if(border === 'left') {
        let [left, right] = [0, nums.length - 1];
        let middle = parseInt((left + right) / 2);
        while(left <= right) {
            if(target < nums[middle]) right = middle - 1;
            else if(target > nums[middle]) left = middle + 1;
            else {
                borderIndex = middle;
                right = middle - 1;
            }
            middle = parseInt((left + right) / 2);
        }
    }else {
        let [left, right] = [0, nums.length - 1];
        let middle = parseInt((left + right) / 2);
        while(left <= right) {
            
            if(target < nums[middle]) right = middle - 1;
            else if(target > nums[middle]) left = middle + 1;
            else {
                borderIndex = middle;
                left = middle + 1;
            }
            middle = parseInt((left + right) / 2);
        }
    }
    return borderIndex;
}

 
var searchRange = function(nums, target) {
    if(nums.length === 0) return [-1, -1];
    let [leftBorder, rightBorder] = [findBorder("left", nums, target), findBorder("right", nums, target)];
    return [leftBorder, rightBorder];
};
```
